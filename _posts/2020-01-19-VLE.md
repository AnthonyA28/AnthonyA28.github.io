function getBaseLog(x, y) {
  return Math.log(y) / Math.log(x);
}

function addData(chart, label, data) {
    chart.data.labels.push(label);
    chart.data.datasets.forEach((dataset) => {
      dataset.data.push(data);
    });
    chart.update();
}

function get_float(name) {
    let input = document.getElementById(name).value;
    let x = 0.0;
    if( input == "" ) {
        window.alert(name + " is empty ");
    }
    if( input != "") {
      x = parseFloat(input);
      if (isNaN(x)){
        window.alert(name + " must be a valid number.");
        document.getElementById(name).value = "";
        return;
      }
      else {
        console.log(name + " = " + x);
        return x;
      }
    }
}

function submitted() {
    console.log("submitting");
    let comp2_A   = get_float("comp2_A");
    let comp2_B   = get_float("comp2_B");
    let comp2_C   = get_float("comp2_C");
    let comp1_A   = get_float("comp1_A");
    let comp1_B   = get_float("comp1_B");
    let comp1_C   = get_float("comp1_C");
    let step    = get_float("step");
    let P_total = get_float("P_total");
    if(step < 0.1 ) {
        alert("Temperature interval is too small, defaulting to 0.1");
        step = 0.1;
    }

    console.log("comp2_A: " + comp2_A);
    console.log("comp2_B: " + comp2_B);
    console.log("comp2_C: " + comp2_C);
    console.log("comp1_A: " + comp1_A);
    console.log("comp1_B: " + comp1_B);
    console.log("comp1_C: " + comp1_C);
    console.log("step: " + step);
    console.log("P_total: " + P_total);



    let P_vap_comp1 = 0;
    let P_vap_comp2 = 0;
    let X_comp2 = 0;
    let X_comp1 = 0;
    let Y_comp2 = 0;
    let Y_comp1 = 0;

    xypoints = [];


    let T_BP_comp1 = comp1_B/(comp1_A-getBaseLog(10,P_total))-comp1_C;
    console.log("estimated T_BP_comp1: ", T_BP_comp1);
    let T_BP_comp2 = comp2_B/(comp2_A-getBaseLog(10,P_total))-comp2_C;
    console.log("estimated T_BP_comp2: ", T_BP_comp2);

    let T_max = T_BP_comp1;
    let T_min = T_BP_comp2;
    if( T_BP_comp2 > T_BP_comp1 ) T_max = T_BP_comp2;
    if( T_BP_comp1 < T_BP_comp2 ) T_min = T_BP_comp1;

    let atLastTemp = false;
    for( T = T_max; T >= T_min-step && !atLastTemp ; T-=step ) {
        if(T < T_min ){ T = T_min; atLastTemp = true;} ;
        // console.log("T: " + T);
        // Antointes equations
        P_vap_comp1 = Math.pow(10,comp1_A-comp1_B/(comp1_C+T));
        P_vap_comp2 = Math.pow(10,comp2_A-comp2_B/(comp2_C+T));
        // Daltons Law  -- total pressure is summation of partial pressures
        // P_total = Pp_comp1 + Pp_comp2
        // insert Raults Law -- partial pressure is equal to vapor pressure times mole frac
        // P_total = P_vap_comp1 * X_comp1 + P_vap_comp2 * X_comp2
        // P_total = P_vap_comp1 * (1-comp2) + P_vap_comp2 * X_comp2
        // Then do some algebra to get
        X_comp2 = (P_total-P_vap_comp1)/(P_vap_comp2-P_vap_comp1)
        X_comp1 = 1 - X_comp2;
        // now use raults law again
        Y_comp2 = P_vap_comp2*X_comp2/P_total;
        Y_comp1 = 1-Y_comp2;
        // if( ( X_comp2 < 1.0 && X_comp2 > 0.0) && ( X_comp1 < 1.0 && X_comp1 > 0.0) && ( Y_comp2 < 1.0 && Y_comp2 > 0.0) && ( Y_comp1 < 1.0 && Y_comp1 > 0.0) ) {

            xypoints.push({x:X_comp2, y:Y_comp2});
        // }
    }
    console.log(xypoints);
